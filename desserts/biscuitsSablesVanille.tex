\preparationtime{15}

\macerationtime{10}

\cookingtime{20}

\totaltime{}

\portions{20}

\begin{ingredients}
    \item 1/2 tasse fécule de maïs
    \item 1/2 tasse sucre à glacer
    \item 1 tasse farine tout usage
    \item 3/4 tasse beurre ramolli
    \item Vanille au goût (facultatif)
\end{ingredients}

\begin{steps}
    \item Tamiser la farine, le sucre à glacer et la fécule de maïs.
    \item Ajouter graduellement le beurre ramolli et la vanille (pas trop) tout en mélangeant, afin d’obtenir une pâte bien lisse et souple.
    \item Avec la pâte, faire des boules d’environ 1 pouce de diamètre et les distancer (5 cm) sur une plaque à biscuits non graissée.
    \item Aplatir les boules et les décorer au choix.
    \item Mettre au four à 350°F de 15 à 20 minutes, ou jusqu’à ce que ce soit légèrement doré sur les côtés.
    \item Laisser reposer quelques minutes et servir avec un bon verre de lait.
\end{steps}
