\preparationtime{5}

\cookingtime{5}

\totaltime{}

\portions{6}

\begin{ingredients}
    \item 3 œufs
    \item 1 tasse lait
    \item Un peu cannelle
    \item Tranches pain
    \item Sirop d'érable
\end{ingredients}

\begin{steps}
    \item Brasser les oeufs, le lait et la cannelle ; y faire tremper le pain.
    \item Faire dorer dans un beurre chaud à feu moyen servir le pain doré avec le sirop.
\end{steps}
